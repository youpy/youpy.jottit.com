<script>
Array.prototype.forEach = function(f, thisObj) {
  for (var i=0, length=this.length; i < length; i++) {
    f.call(thisObj, this[i], i, this);
  }
};
function qw(str) {
  return str.split(/ /);
}
qw('wrapper').forEach(function(elm_id) {
  var elm = document.getElementById(elm_id);
  elm.innerHTML = '';
});
</script>
